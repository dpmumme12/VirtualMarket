### Clears all images and build cache ###
docker system prune -a

### Build the images and starts containers ###
docker-compose -f docker-compose.yml up -d --build

### Running django command against the web service to collectic static files ### 
docker-compose -f docker-compose.yml exec web python3 manage.py collectstatic --no-input --clear

### see all running containers ###
docker-compose ps

### Shutdown and remove all socker conatiners ##

docker rm $(docker ps -a -q)
